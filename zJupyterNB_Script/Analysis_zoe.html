def add(df):
    approp = df['WaterUsagePerAcre']*12/df['Precip']
    df['Appropriate'] = pd.Series(approp,index = df.index)
    df = df.sort_values('Appropriate', axis=0)
    df = df.reset_index(drop=True)
    return df
